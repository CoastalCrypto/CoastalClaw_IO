# Plan 2: PRs + Time-Travel Snapshots

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the architect's loop end-to-end: after a green build, create a PR via `gh`, poll for merge, and capture Time-Travel snapshots at key moments so the operator can restore any prior state.

**Architecture:** Three new pure-function stages (`pr-creation`, `pr-review`, `snapshot-capture`) plug into the existing `stage-runner.ts` orchestrator. A `SnapshotManager` handles the shadow-git repo. A `ModelRouterAdapter` bridges the real `ModelRouter` to the architect's `ModelRouterLike` interface. The daemon gains a two-phase tick: poll active PRs, then process pending items. An `architect_events` table provides the seam for Plan 3's SSE layer.

**Tech Stack:** TypeScript, ESM, Node 22+, pnpm monorepo, better-sqlite3, vitest, `gh` CLI, `git` CLI

```ts
// packages/architect/src/event-log.ts
import type Database from 'better-sqlite3'
import { ulid } from '@coastal-ai/core/architect/ulid'

export interface ArchitectEvent {
  id: string
  cycleId: string | null
  workItemId: string | null
  eventType: string
  payload: Record<string, unknown> | null
  createdAt: number
}

export class EventLog {
  constructor(private db: Database.Database) {}

  emit(
    eventType: string,
    opts: { cycleId?: string | null; workItemId?: string | null; [key: string]: unknown },
  ): void {
    const { cycleId, workItemId, ...rest } = opts
    const id = ulid()
    const payload = Object.keys(rest).length > 0 ? JSON.stringify(rest) : null
    this.db.prepare(`
      INSERT INTO architect_events (id, cycle_id, work_item_id, event_type, payload, created_at)
      VALUES (?, ?, ?, ?, ?, ?)
    `).run(id, cycleId ?? null, workItemId ?? null, eventType, payload, Date.now())
  }

  listForWorkItem(workItemId: string, limit = 100): ArchitectEvent[] {
    const rows = this.db.prepare(
      'SELECT * FROM architect_events WHERE work_item_id = ? ORDER BY created_at ASC LIMIT ?'
    ).all(workItemId, limit) as any[]
    return rows.map(r => this.fromRow(r))
  }

  listSince(sinceMs: number, limit = 100): ArchitectEvent[] {
    const rows = this.db.prepare(
      'SELECT * FROM architect_events WHERE created_at >= ? ORDER BY created_at ASC LIMIT ?'
    ).all(sinceMs, limit) as any[]
    return rows.map(r => this.fromRow(r))
  }

  private fromRow(r: any): ArchitectEvent {
    return {
      id: r.id,
      cycleId: r.cycle_id,
      workItemId: r.work_item_id,
      eventType: r.event_type,
      payload: r.payload ? JSON.parse(r.payload) : null,
      createdAt: r.created_at,
    }
  }
}
```

```ts
// packages/architect/src/snapshots.ts
import { execFileSync } from 'node:child_process'
import { existsSync, mkdirSync } from 'node:fs'
import { join } from 'node:path'
import type Database from 'better-sqlite3'
import { ulid } from '@coastal-ai/core/architect/ulid'

export interface SnapshotRecord {
  id: string
  cycleId: string | null
  workItemId: string | null
  shadowRef: string
  parentId: string | null
  capturedAt: number
  capturedBy: string
  note: string | null
  retention: 'short' | 'long' | 'pinned'
}

export interface SnapshotManagerOpts {
  repoRoot: string
  dataDir: string
  db: Database.Database
}

export class SnapshotManager {
  private readonly shadowDir: string
  private readonly repoRoot: string
  private readonly db: Database.Database

  constructor(opts: SnapshotManagerOpts) {
    this.repoRoot = opts.repoRoot
    this.shadowDir = join(opts.dataDir, '.architect-snapshots')
    this.db = opts.db
    this.ensureShadowRepo()
  }

  private ensureShadowRepo(): void {
    const gitDir = join(this.shadowDir, '.git')
    if (!existsSync(gitDir)) {
      mkdirSync(this.shadowDir, { recursive: true })
      execFileSync('git', ['init', '--bare', join(this.shadowDir, '.git')], { stdio: 'pipe' })
    }
  }

  private gitShadow(args: string[]): string {
    return execFileSync('git', args, {
      cwd: this.repoRoot,
      encoding: 'utf8',
      env: {
        ...process.env,
        GIT_DIR: join(this.shadowDir, '.git'),
        GIT_WORK_TREE: this.repoRoot,
      },
      stdio: ['pipe', 'pipe', 'pipe'],
    }).trim()
  }

  capture(opts: {
    cycleId: string | null
    workItemId: string | null
    capturedBy: string
    note?: string
    retention?: 'short' | 'long' | 'pinned'
  }): SnapshotRecord {
    // Write workspace tree to shadow repo
    this.gitShadow(['add', '-A'])
    const treeHash = this.gitShadow(['write-tree'])

    // Find parent (latest snapshot)
    const lastSnap = this.db.prepare(
      'SELECT id, shadow_ref FROM snapshots ORDER BY captured_at DESC LIMIT 1'
    ).get() as { id: string; shadow_ref: string } | undefined

    // Create commit-tree (parented if we have a prior snapshot)
    const parentArgs = lastSnap ? ['-p', lastSnap.shadow_ref] : []
    const commitHash = this.gitShadow([
      'commit-tree', treeHash, ...parentArgs, '-m',
      `snapshot ${opts.capturedBy} at ${new Date().toISOString()}`,
    ])

    const id = ulid()
    const now = Date.now()
    this.db.prepare(`
      INSERT INTO snapshots (id, cycle_id, work_item_id, shadow_ref, parent_id, captured_at, captured_by, note, retention)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    `).run(
      id, opts.cycleId ?? null, opts.workItemId ?? null,
      commitHash, lastSnap?.id ?? null, now,
      opts.capturedBy, opts.note ?? null, opts.retention ?? 'short',
    )

    return {
      id, cycleId: opts.cycleId ?? null, workItemId: opts.workItemId ?? null,
      shadowRef: commitHash, parentId: lastSnap?.id ?? null,
      capturedAt: now, capturedBy: opts.capturedBy,
      note: opts.note ?? null, retention: opts.retention ?? 'short',
    }
  }

  restore(snapshotId: string): string {
    const snap = this.db.prepare('SELECT * FROM snapshots WHERE id = ?').get(snapshotId) as any
    if (!snap) throw new Error(`Snapshot ${snapshotId} not found`)

    const branchName = `architect/restore-${snapshotId.slice(-8)}`

    // Read the tree from the shadow commit
    const treeHash = this.gitShadow(['rev-parse', `${snap.shadow_ref}^{tree}`])

    // Create restore branch in real repo and set it to the snapshot's tree
    execFileSync('git', ['checkout', '-b', branchName], { cwd: this.repoRoot, stdio: 'pipe' })
    execFileSync('git', ['read-tree', '--reset', '-u', treeHash], {
      cwd: this.repoRoot,
      env: {
        ...process.env,
        GIT_DIR: join(this.repoRoot, '.git'),
        GIT_WORK_TREE: this.repoRoot,
        GIT_INDEX_FILE: join(this.shadowDir, '.git', 'restore-index'),
      },
      stdio: 'pipe',
    })
    // Actually: simpler approach — checkout the tree from shadow into real working dir
    // Use git archive from shadow, extract into real repo, then commit
    execFileSync('git', ['checkout', '-b', branchName], { cwd: this.repoRoot, stdio: 'pipe' }).toString()

    // Pipe: archive from shadow → extract to working dir → commit in real repo
    const archiveData = execFileSync('git', ['archive', snap.shadow_ref], {
      cwd: this.repoRoot,
      env: { ...process.env, GIT_DIR: join(this.shadowDir, '.git') },
      maxBuffer: 50 * 1024 * 1024,
    })
    // Write archive to temp file and extract
    const archivePath = join(this.shadowDir, 'restore.tar')
    const { writeFileSync: wfs, unlinkSync: uls } = require('node:fs')
    wfs(archivePath, archiveData)
    execFileSync('tar', ['xf', archivePath], { cwd: this.repoRoot, stdio: 'pipe' })
    try { uls(archivePath) } catch {}

    execFileSync('git', ['add', '-A'], { cwd: this.repoRoot, stdio: 'pipe' })
    execFileSync('git', ['commit', '-m', `revert(architect): restore workspace to ${snap.captured_at}`], {
      cwd: this.repoRoot, stdio: 'pipe',
    })

    return branchName
  }

  listForCycle(cycleId: string): SnapshotRecord[] {
    return (this.db.prepare(
      'SELECT * FROM snapshots WHERE cycle_id = ? ORDER BY captured_at ASC'
    ).all(cycleId) as any[]).map(r => this.fromRow(r))
  }

  pin(snapshotId: string): void {
    this.db.prepare('UPDATE snapshots SET retention = ? WHERE id = ?').run('pinned', snapshotId)
  }

  prune(opts: { shortMaxAgeMs?: number; longMaxAgeMs?: number } = {}): number {
    const now = Date.now()
    const shortCutoff = now - (opts.shortMaxAgeMs ?? 30 * 24 * 60 * 60 * 1000)
    const longCutoff = now - (opts.longMaxAgeMs ?? 365 * 24 * 60 * 60 * 1000)

    const result = this.db.prepare(`
      DELETE FROM snapshots
      WHERE (retention = 'short' AND captured_at < ?)
         OR (retention = 'long' AND captured_at < ?)
    `).run(shortCutoff, longCutoff)

    return result.changes
  }

  private fromRow(r: any): SnapshotRecord {
    return {
      id: r.id,
      cycleId: r.cycle_id,
      workItemId: r.work_item_id,
      shadowRef: r.shadow_ref,
      parentId: r.parent_id,
      capturedAt: r.captured_at,
      capturedBy: r.captured_by,
      note: r.note,
      retention: r.retention,
    }
  }
}
```

The test uses a real temp git repo so it will validate the actual mechanism.

```ts
async pushBranch(branchName: string): Promise<void> {
  execFileSync('git', ['push', '-u', 'origin', branchName], { cwd: this.repoRoot, stdio: 'pipe' })
}

branchExistsOnRemote(branchName: string): boolean {
  try {
    const result = execFileSync('git', ['ls-remote', '--heads', 'origin', branchName], {
      cwd: this.repoRoot, encoding: 'utf8', stdio: 'pipe',
    })
    return result.trim().length > 0
  } catch {
    return false
  }
}
```

```ts
// packages/architect/src/stages/pr-creation.ts
import type { WorkItem, Cycle } from '@coastal-ai/core/architect/types'

export interface PRCreationInput {
  workItem: WorkItem
  cycle: Cycle
  branchName: string
  planText: string
  diffText: string
  testSummary: string
  modelUsed: string
  commitAndPush: (message: string, branchName: string) => Promise<void>
  createPR: (opts: {
    title: string
    body: string
    branchName: string
    draft: boolean
  }) => Promise<{ prUrl: string; prNumber: number }>
  commentOnIssue?: (issueRef: string, comment: string) => Promise<void>
}

export type PRCreationResult =
  | { kind: 'ok'; prUrl: string; prNumber: number }
  | { kind: 'hard_fail'; failureKind: 'env_push' | 'env_gh'; message: string }

export async function runPRCreationStage(input: PRCreationInput): Promise<PRCreationResult> {
  const { workItem, cycle, branchName, planText, testSummary, modelUsed } = input

  // 1. Commit and push
  const commitMsg = [
    `chore(architect): ${workItem.title}`,
    '',
    `Work item: ${workItem.id}`,
    `Iteration: ${cycle.iteration}`,
    '',
    `Co-Authored-By: coastal-architect <bot@coastal.ai>`,
    `Architect-Cycle: ${cycle.id}`,
  ].join('\n')

  try {
    await input.commitAndPush(commitMsg, branchName)
  } catch (err: unknown) {
    const message = err instanceof Error ? err.message : String(err)
    return { kind: 'hard_fail', failureKind: 'env_push', message }
  }

  // 2. Create PR
  const prBody = buildPRBody({ workItem, cycle, planText, testSummary, modelUsed })
  const draft = workItem.approvalPolicy === 'full'

  try {
    const { prUrl, prNumber } = await input.createPR({
      title: `[architect] ${workItem.title}`,
      body: prBody,
      branchName,
      draft,
    })

    // 3. Backlink to source issue (if applicable)
    if (workItem.source === 'github' && workItem.sourceRef && input.commentOnIssue) {
      try {
        await input.commentOnIssue(
          workItem.sourceRef,
          `Architect opened PR #${prNumber}: ${prUrl}`,
        )
      } catch { /* non-fatal */ }
    }

    return { kind: 'ok', prUrl, prNumber }
  } catch (err: unknown) {
    const message = err instanceof Error ? err.message : String(err)
    return { kind: 'hard_fail', failureKind: 'env_gh', message }
  }
}

function buildPRBody(opts: {
  workItem: WorkItem
  cycle: Cycle
  planText: string
  testSummary: string
  modelUsed: string
}): string {
  const { workItem, cycle, planText, testSummary, modelUsed } = opts
  return [
    `## Plan`,
    '',
    planText,
    '',
    `## Test Summary`,
    '',
    testSummary,
    '',
    `## Metadata`,
    '',
    `| Field | Value |`,
    `|---|---|`,
    `| Work Item | \`${workItem.id}\` |`,
    `| Source | ${workItem.source} |`,
    `| Iteration | ${cycle.iteration} |`,
    `| Model | ${modelUsed} |`,
    `| Cycle | \`${cycle.id}\` |`,
    '',
    `---`,
    `*Generated by coastal-architect*`,
  ].join('\n')
}
```

```ts
// packages/architect/src/model-router-adapter.ts
import type { ModelRouterLike, ModelDescriptor, Tier } from './model-router-client.js'

interface RealModelRouter {
  chat(
    messages: Array<{ role: string; content: string }>,
    options?: { model?: string },
  ): Promise<{ reply: string; decision: any }>
  cascade: {
    route(message: string): Promise<{
      model: string | null
      domain: string
      urgency: string
      fallbackModels: string[]
    }>
  }
}

const URGENCY_TO_TIER: Record<string, Tier> = {
  high: 'apex',
  medium: 'standard',
  low: 'lite',
}

export function createModelRouterAdapter(router: RealModelRouter): ModelRouterLike {
  return {
    async getModelFor(
      _domain: string,
      priority: 'low' | 'medium' | 'high',
    ): Promise<ModelDescriptor | null> {
      // Route a synthetic message to discover which model the cascade picks
      const decision = await router.cascade.route(
        `[architect:${priority}] routing probe`,
      )
      if (!decision.model) return null
      const tier = URGENCY_TO_TIER[decision.urgency] ?? 'lite'
      return { id: decision.model, tier }
    },

    async generate(modelId: string, prompt: string): Promise<string> {
      const { reply } = await router.chat(
        [{ role: 'user', content: prompt }],
        { model: modelId },
      )
      return reply
    },
  }
}
```

```ts
async tick(): Promise<'ran' | 'idle' | 'locked' | 'polled'> {
  if (this.locked) return 'locked'
  this.locked = true
  try {
    // Phase 1: Poll active PRs
    const polled = await this.pollActivePRs()
    // Phase 2: Process next pending item
    const next = this.deps.workStore.listPending(1)[0]
    if (!next) return polled ? 'polled' : 'idle'
    await runWorkItemCycle({ ... })
    return 'ran'
  } finally {
    this.locked = false
  }
}

private async pollActivePRs(): Promise<boolean> {
  if (!this.deps.pollPR) return false
  // Find cycles in pr_review stage
  const reviewing = this.deps.cycleStore.listByStage('pr_review')
  let anyPolled = false
  for (const cycle of reviewing) {
    if (!cycle.prUrl) continue
    const result = await this.deps.pollPR(cycle.prUrl)
    if (result.status === 'merged') {
      this.deps.cycleStore.terminate(cycle.id, { outcome: 'merged' })
      if (cycle.workItemId) {
        this.deps.workStore.updateStatus(cycle.workItemId, 'merged')
      }
      this.deps.emitEvent?.('pr_merged', { cycleId: cycle.id, workItemId: cycle.workItemId })
      anyPolled = true
    } else if (result.status === 'closed') {
      this.deps.cycleStore.terminate(cycle.id, { outcome: 'vetoed' })
      if (cycle.workItemId) {
        this.deps.workStore.updateStatus(cycle.workItemId, 'cancelled')
      }
      this.deps.emitEvent?.('pr_closed', { cycleId: cycle.id, workItemId: cycle.workItemId })
      anyPolled = true
    }
  }
  return anyPolled
}
```
