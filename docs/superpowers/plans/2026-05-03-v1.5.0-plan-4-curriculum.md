# Plan 4: Curriculum Mode — Idle-Time Self-Improvement

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When the work queue is empty, the architect scans the codebase for improvement opportunities and self-generates low-priority work items — with safety rails, suppression, and budget limits.

**Architecture:** A `CurriculumScanner` class harvests codebase signals (stale TODOs, churn, lint warnings), makes one LLM call to propose work items, validates proposals against locked-paths and dedup, inserts valid ones into `work_items` with `source='curriculum'`. The daemon's idle path calls this scanner. A `SuppressionStore` tracks rejected proposals to prevent re-proposal for 30 days.

**Tech Stack:** TypeScript, ESM, Node 22+, better-sqlite3, vitest, `execFileSync` for git/grep

**Spec:** `docs/superpowers/specs/2026-04-29-v1.5.0-self-healing-release-design.md` section 3 (Curriculum mode)

// packages/architect/src/curriculum/suppression-store.ts
import type Database from 'better-sqlite3'
import { ulid } from '@coastal-ai/core/architect/ulid'

const THIRTY_DAYS_MS = 30 * 24 * 60 * 60 * 1000

export class SuppressionStore {
  constructor(private db: Database.Database) {}

  static buildSignature(title: string, targetHints: string[] | null): string {
    const normalTitle = title.toLowerCase().trim()
    const hints = (targetHints ?? []).slice().sort().join(',')
    return hints ? `${normalTitle}|${hints}` : normalTitle
  }

  isSuppressed(signature: string): boolean {
    const row = this.db.prepare(
      'SELECT id FROM curriculum_suppressions WHERE signature = ? AND suppressed_until > ?'
    ).get(signature, Date.now())
    return !!row
  }

  suppress(signature: string, reason: 'vetoed' | 'failed' | 'duplicate'): void {
    const id = ulid()
    this.db.prepare(`
      INSERT INTO curriculum_suppressions (id, signature, suppressed_until, reason, created_at)
      VALUES (?, ?, ?, ?, ?)
    `).run(id, signature, Date.now() + THIRTY_DAYS_MS, reason, Date.now())
  }

  pruneExpired(): number {
    return this.db.prepare(
      'DELETE FROM curriculum_suppressions WHERE suppressed_until <= ?'
    ).run(Date.now()).changes
  }
}
```

// packages/architect/src/curriculum/signals.ts
export interface TodoSignal { file: string; line: number; text: string }
export interface ChurnSignal { file: string; changes: number }

export function findStaleTodos(grepOutput: string): TodoSignal[] {
  if (!grepOutput.trim()) return []
  return grepOutput.trim().split('\n').map(line => {
    const match = line.match(/^(.+?):(\d+):\s*(.+)$/)
    if (!match) return null
    return { file: match[1], line: Number(match[2]), text: match[3].trim() }
  }).filter((x): x is TodoSignal => x !== null)
}

export function findChurnHotspots(gitLogOutput: string, minChanges = 3): ChurnSignal[] {
  if (!gitLogOutput.trim()) return []
  const counts = new Map<string, number>()
  for (const file of gitLogOutput.trim().split('\n')) {
    const f = file.trim()
    if (!f) continue
    counts.set(f, (counts.get(f) ?? 0) + 1)
  }
  return [...counts.entries()]
    .filter(([, cnt]) => cnt >= minChanges)
    .sort((a, b) => b[1] - a[1])
    .map(([file, changes]) => ({ file, changes }))
}

export interface HarvestedSignals {
  staleTodos: TodoSignal[]
  churnHotspots: ChurnSignal[]
}

export function harvestSignals(opts: {
  execCommand: (cmd: string, args: string[]) => string
  repoRoot: string
}): HarvestedSignals {
  let todoOutput = ''
  try {
    todoOutput = opts.execCommand('grep', ['-rnE', 'TODO|FIXME|XXX', '--include=*.ts', '--include=*.tsx', 'packages/'])
  } catch { /* grep returns exit 1 on no match */ }

  let churnOutput = ''
  try {
    churnOutput = opts.execCommand('git', ['log', '--since=30 days ago', '--name-only', '--pretty=format:'])
  } catch { /* empty repo or no commits in range */ }

  return {
    staleTodos: findStaleTodos(todoOutput),
    churnHotspots: findChurnHotspots(churnOutput),
  }
}
```

- [ ] **Step 2: Implement CurriculumScanner**

```ts
// packages/architect/src/curriculum/scanner.ts
import type { WorkItemStore } from '@coastal-ai/core/architect/store'
import type { CycleStore } from '@coastal-ai/core/architect/cycle-store'
import type { SuppressionStore } from './suppression-store.js'
import type { HarvestedSignals } from './signals.js'
import { existsSync } from 'node:fs'
import { join } from 'node:path'

export interface ScannerDeps {
  workStore: WorkItemStore
  cycleStore: CycleStore
  suppressions: SuppressionStore
  repoRoot: string
  callLLM: (prompt: string) => Promise<{ text: string; modelId: string }>
  harvestSignals: () => HarvestedSignals
  isLockedPath: (path: string) => string | null
  dailyBudget: number
}

export interface ScanResult {
  proposalsInserted: number
  proposalsRejected: number
  signalsSummarized: number
  modelUsed: string
}

interface Proposal {
  title: string
  targetHints: string[]
  budgetLoc: number
}

export class CurriculumScanner {
  constructor(private deps: ScannerDeps) {}

  async scan(): Promise<ScanResult> {
    const { workStore, cycleStore, suppressions, repoRoot } = this.deps

    // Record scan cycle
    const cycle = cycleStore.start(null as any, { kind: 'curriculum_scan' })

    // 1. Harvest signals
    const signals = this.deps.harvestSignals()
    const signalCount = signals.staleTodos.length + signals.churnHotspots.length

    // 2. Build prompt
    const prompt = this.buildPrompt(signals)

    // 3. Call LLM
    let proposals: Proposal[] = []
    let modelUsed = 'unknown'
    try {
      const resp = await this.deps.callLLM(prompt)
      modelUsed = resp.modelId
      proposals = this.parseProposals(resp.text)
    } catch {
      cycleStore.terminate(cycle.id, { outcome: 'failed', modelUsed, errorMessage: 'LLM call failed' })
      return { proposalsInserted: 0, proposalsRejected: 0, signalsSummarized: signalCount, modelUsed }
    }

    // 4. Validate + insert
    let inserted = 0
    let rejected = 0

    for (const p of proposals.slice(0, this.deps.dailyBudget)) {
      // Check locked paths
      const locked = p.targetHints.find(h => this.deps.isLockedPath(h))
      if (locked) { rejected++; continue }

      // Check file exists
      const fileExists = p.targetHints.some(h => existsSync(join(repoRoot, h)))
      if (!fileExists && p.targetHints.length > 0) { rejected++; continue }

      // Check suppression
      const sig = SuppressionStore.buildSignature(p.title, p.targetHints)
      if (suppressions.isSuppressed(sig)) { rejected++; continue }

      // Check dedup (existing pending/active item)
      try {
        workStore.insert({
          source: 'curriculum',
          title: p.title,
          body: `Curriculum-proposed: ${p.title}`,
          targetHints: p.targetHints,
          budgetLoc: Math.min(p.budgetLoc, 200),
          approvalPolicy: 'full',
          priority: 'low',
          allowSelfModify: false,
        })
        inserted++
      } catch {
        // DedupConflictError or other → skip
        rejected++
      }
    }

    const outcome = inserted > 0 ? 'merged' : 'failed'
    cycleStore.terminate(cycle.id, {
      outcome: outcome as any,
      modelUsed,
      testSummary: JSON.stringify({ proposalsInserted: inserted, proposalsRejected: rejected, signalsSummarized: signalCount }),
    })

    return { proposalsInserted: inserted, proposalsRejected: rejected, signalsSummarized: signalCount, modelUsed }
  }

  private buildPrompt(signals: HarvestedSignals): string {
    const todoSection = signals.staleTodos.length > 0
      ? `STALE TODOs:\n${signals.staleTodos.slice(0, 20).map(t => `  ${t.file}:${t.line} ${t.text}`).join('\n')}`
      : 'No stale TODOs found.'
    const churnSection = signals.churnHotspots.length > 0
      ? `CHURN HOTSPOTS (30d):\n${signals.churnHotspots.slice(0, 10).map(c => `  ${c.file} (${c.changes} changes)`).join('\n')}`
      : 'No churn hotspots.'

    return `You are the Coastal.AI Curriculum Scanner. Based on the codebase signals below, propose 0-5 small, focused improvements as JSON.

SIGNALS:
${todoSection}

${churnSection}

RULES:
- Only propose changes you have specific evidence for above.
- Each proposal must reference a real file from the signals.
- Keep budgetLoc under 200.
- No architectural changes. Only bug fixes, cleanup, and small refactors.
- Do NOT propose changes to packages/architect/ (self-modify guard).

OUTPUT: JSON array of objects with { title, targetHints: string[], budgetLoc: number }. Return [] if nothing actionable.`
  }

  private parseProposals(text: string): Proposal[] {
    try {
      const match = text.match(/\[[\s\S]*\]/)
      if (!match) return []
      const parsed = JSON.parse(match[0])
      if (!Array.isArray(parsed)) return []
      return parsed.filter((p: any) =>
        typeof p.title === 'string' &&
        Array.isArray(p.targetHints) &&
        typeof p.budgetLoc === 'number'
      )
    } catch {
      return []
    }
  }
}
```

If not, pass an empty string or modify the method to accept `string | null`. The DB column allows NULL.
