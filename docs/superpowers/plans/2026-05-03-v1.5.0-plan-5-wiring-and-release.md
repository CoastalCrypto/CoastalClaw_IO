```ts
// Model router
import { ModelRouter } from '@coastal-ai/core/models/router'
import { createModelRouterAdapter } from './model-router-adapter.js'
import { ArchitectModelRouterClient } from './model-router-client.js'
import { runPlanningStage } from './stages/planning.js'
import { runBuildingStage } from './stages/building.js'
import { runPRCreationStage } from './stages/pr-creation.js'
import { pollPRStatus, triggerAutoMerge } from './stages/pr-review.js'
import { SnapshotManager } from './snapshots.js'
import { EventLog } from './event-log.js'
import { CurriculumScanner } from './curriculum/scanner.js'
import { SuppressionStore } from './curriculum/suppression-store.js'
import { harvestSignals } from './curriculum/signals.js'
import { loadWorkspaceMapSync } from './workspace-map.js'
import { findTouchedPackages } from './touched-packages.js'
import { runLintGate, runTypeGate, runBuildGate, runTestGate } from './gates.js'
import { isLockedPath } from './config.js'
import { execFileSync } from 'node:child_process'

// ... inside the non-legacy block:
const router = new ModelRouter({ ollamaUrl: OLLAMA_URL, defaultModel: OLLAMA_MODEL })
const adapter = createModelRouterAdapter(router)
const client = new ArchitectModelRouterClient(adapter, {
  minTier: (process.env.CC_ARCHITECT_MIN_TIER as any) ?? 'low',
})
const patcher = new Patcher(REPO_ROOT)
const workspaceMap = loadWorkspaceMapSync(REPO_ROOT)
const snapshotMgr = new SnapshotManager({ repoRoot: REPO_ROOT, dataDir: DATA_DIR, db: architectDb })
const eventLog = new EventLog(architectDb)
const suppressions = new SuppressionStore(architectDb)

const daemon = new ArchitectDaemon({
  workStore, cycleStore,
  runPlan: async (input) => {
    return runPlanningStage({
      workItem: input.workItem,
      reviseContext: input.reviseContext,
      readSourceFile: async (relPath) => readFileSync(join(REPO_ROOT, relPath), 'utf8'),
      client,
      lockedPathCheck: (p) => isLockedPath(p),
    })
  },
  runBuild: async (input) => {
    const exec = async (cmd: string, opts: { cwd: string }) => {
      try {
        const stdout = execFileSync('sh', ['-c', cmd], { cwd: opts.cwd, encoding: 'utf8', stdio: 'pipe' })
        return { stdout, stderr: '', exitCode: 0 }
      } catch (err: any) {
        return { stdout: err.stdout ?? '', stderr: err.stderr ?? '', exitCode: err.status ?? 1 }
      }
    }
    await patcher.createBranch(input.branchName)
    await patcher.applyDiff(input.diff)
    const touched = findTouchedPackages(input.diff, workspaceMap)
    const gateOpts = { cwd: REPO_ROOT, exec }
    return runBuildingStage({
      diff: input.diff,
      applyDiff: async () => {}, // already applied above
      runLint: () => runLintGate(touched, gateOpts),
      runTypecheck: () => runTypeGate(touched, gateOpts),
      runBuild: () => runBuildGate(touched, gateOpts),
      runTests: () => runTestGate(touched, gateOpts),
    })
  },
  runPR: async (input) => {
    return runPRCreationStage({
      ...input,
      commitAndPush: async (msg, branch) => {
        await patcher.commitChange(msg)
        await patcher.pushBranch(branch)
      },
      createPR: async (opts) => {
        const output = execFileSync('gh', [
          'pr', 'create',
          '--title', opts.title,
          '--body', opts.body,
          '--head', opts.branchName,
          ...(opts.draft ? ['--draft'] : []),
        ], { cwd: REPO_ROOT, encoding: 'utf8', stdio: 'pipe' })
        const prUrl = output.trim()
        const prNumber = Number(prUrl.split('/').pop()) || 0
        return { prUrl, prNumber }
      },
    })
  },
  pollPR: async (prUrl) => {
    try {
      const json = execFileSync('gh', ['pr', 'view', prUrl, '--json', 'state,mergedAt'], {
        encoding: 'utf8', stdio: 'pipe',
      })
      const pr = JSON.parse(json)
      return pollPRStatus({ prUrl, ghView: async () => pr })
    } catch {
      return { status: 'error', message: 'gh pr view failed' }
    }
  },
  autoMerge: async (prUrl) => {
    return triggerAutoMerge({
      prUrl,
      ghMerge: async (url) => {
        execFileSync('gh', ['pr', 'merge', url, '--squash', '--auto'], { stdio: 'pipe' })
      },
    })
  },
  captureSnapshot: (opts) => {
    snapshotMgr.capture(opts)
  },
  emitEvent: (type, payload) => {
    eventLog.emit(type, payload)
  },
  isApprovalRequired: (gate) => {
    // Read mode file to determine
    const modeFile = join(DATA_DIR, '.architect-mode')
    let mode = 'hands-on'
    try { mode = JSON.parse(readFileSync(modeFile, 'utf8')).mode } catch {}
    if (mode === 'autopilot') return false
    if (mode === 'hands-off' && gate === 'plan') return false
    return true // hands-on requires all gates
  },
  curriculumScanner: new CurriculumScanner({
    workStore, cycleStore, suppressions, repoRoot: REPO_ROOT,
    callLLM: async (prompt) => client.callPlan(prompt),
    harvestSignals: () => harvestSignals({
      execCommand: (cmd, args) => execFileSync(cmd, args, { cwd: REPO_ROOT, encoding: 'utf8', stdio: 'pipe' }),
      repoRoot: REPO_ROOT,
    }),
    isLockedPath: (p) => isLockedPath(p),
    dailyBudget: 1,
  }),
  curriculumEnabled: false, // off by default
  log: (msg) => console.log(`[coastal-architect] ${msg}`),
})
```
