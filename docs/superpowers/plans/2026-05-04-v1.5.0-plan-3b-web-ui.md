### api/architect.ts — typed wrapper around coreClient

```ts
import { coreClient } from './client.js'

export interface WorkItem {
  id: string; title: string; body: string; status: string; source: string
  priority: string; approvalPolicy: string; createdAt: number; updatedAt: number
}

export interface Cycle {
  id: string; workItemId: string | null; iteration: number; stage: string
  outcome: string | null; planText: string | null; diffText: string | null
  prUrl: string | null; testSummary: string | null; modelUsed: string | null
  failureKind: string | null; createdAt: number; updatedAt: number
}

export interface ArchitectStatus { power: string; mode: string }

export interface Insights {
  successRate: number; avgIterations: number; totalDurationMs: number
  topFailureKind: string | null; openQueueDepth: number; errorCount: number
}

export const architectApi = {
  getStatus: () => coreClient.get<ArchitectStatus>('/api/admin/architect/status'),
  getWorkItems: (status = 'all') => coreClient.get<WorkItem[]>(`/api/admin/architect/work-items?status=${status}`),
  createWorkItem: (data: { title: string; body?: string }) => coreClient.post<WorkItem>('/api/admin/architect/work-items', data),
  patchWorkItem: (id: string, data: Record<string, unknown>) => coreClient.patch<WorkItem>(`/api/admin/architect/work-items/${id}`, data),
  getActivity: (limit = 50) => coreClient.get<Cycle[]>(`/api/admin/architect/activity?limit=${limit}`),
  getCycle: (id: string) => coreClient.get<Cycle>(`/api/admin/architect/cycles/${id}`),
  postApproval: (id: string, data: { gate: string; decision: string; comment?: string }) =>
    coreClient.post(`/api/admin/architect/cycles/${id}/approval`, data),
  setPower: (state: 'on' | 'off') => coreClient.post('/api/admin/architect/power', { state }),
  setMode: (mode: string) => coreClient.post('/api/admin/architect/mode', { mode }),
  runNow: () => coreClient.post('/api/admin/architect/run-now', {}),
  getInsights: (range = 30) => coreClient.get<Insights>(`/api/admin/architect/insights?range=${range}`),
  getReceipts: () => coreClient.get<{ prs: any[]; totals: { prsMerged: number } }>('/api/admin/architect/receipts'),
}
```

**IMPORTANT:** The implementer MUST read `api/client.ts` first to verify that `coreClient` has `.get()`, `.post()`, `.patch()` methods. If not (if it uses different method names like `.fetch()` or custom patterns), adapt the calls. The existing Channels.tsx shows the actual call pattern.

## Chunk 2: Main Architect page with status card

**Files:**
- Create: `packages/web/src/pages/Architect.tsx`

The main page shows:
1. **Status card** — "What's happening right now" (idle/working/waiting/paused/off)
2. **Tab navigation** — Queue | Activity | Insights | Receipts | Settings
3. Renders the selected sub-page component

```tsx
// Simplified structure
export function Architect({ onNav }: { onNav: (page: NavPage) => void }) {
  const [tab, setTab] = useState<'queue' | 'activity' | 'insights' | 'receipts' | 'settings'>('queue')
  const [status, setStatus] = useState<ArchitectStatus | null>(null)

  useEffect(() => { architectApi.getStatus().then(setStatus).catch(() => {}) }, [])

  return (
    <div className="min-h-screen text-white" style={{ background: '#050a0f' }}>
      <NavBar page={'architect'} onNav={onNav} />
      <div className="pt-20 pb-12 px-4 sm:px-6 max-w-5xl mx-auto">
        <StatusCard status={status} />
        <TabBar tab={tab} setTab={setTab} />
        {tab === 'queue' && <Queue />}
        {tab === 'activity' && <Activity />}
        {tab === 'insights' && <InsightsTab />}
        {tab === 'receipts' && <ReceiptsTab />}
        {tab === 'settings' && <SettingsTab />}
      </div>
    </div>
  )
}
```
