// packages/architect/src/callback-signer.ts
import { createHmac, randomBytes } from 'node:crypto'

export class CallbackSigner {
  private key: Buffer

  constructor(key: Buffer) {
    this.key = key
  }

  static generateKey(): Buffer {
    return randomBytes(32)
  }

  sign(payload: { cycleId: string; gate: string; decision: string; expiresAt: number }): string {
    const data = JSON.stringify(payload)
    const hmac = createHmac('sha256', this.key).update(data).digest('hex')
    const token = Buffer.from(JSON.stringify({ ...payload, hmac })).toString('base64url')
    return token
  }

  verify(token: string): { cycleId: string; gate: string; decision: string; expiresAt: number } | null {
    try {
      const decoded = JSON.parse(Buffer.from(token, 'base64url').toString())
      const { hmac, ...payload } = decoded
      const expected = createHmac('sha256', this.key).update(JSON.stringify(payload)).digest('hex')
      if (hmac !== expected) return null
      if (Date.now() > payload.expiresAt) return null
      return payload
    } catch {
      return null
    }
  }
}
```

// packages/architect/src/notifier.ts
import type { CallbackSigner } from './callback-signer.js'

export interface NotifierDeps {
  broadcast: (message: string) => Promise<void>
  signer: CallbackSigner
  callbackBaseUrl: string
  dashboardBaseUrl: string
}

export class ArchitectNotifier {
  constructor(private deps: NotifierDeps) {}

  async notifyApprovalNeeded(opts: {
    cycleId: string
    gate: 'plan' | 'pr'
    workItemTitle: string
    planSummary: string
    diffStats: string
    reviewTimeoutMin: number
  }): Promise<void> {
    const expiresAt = Date.now() + opts.reviewTimeoutMin * 60 * 1000
    const approveToken = this.deps.signer.sign({ cycleId: opts.cycleId, gate: opts.gate, decision: 'approved', expiresAt })
    const rejectToken = this.deps.signer.sign({ cycleId: opts.cycleId, gate: opts.gate, decision: 'rejected', expiresAt })

    const message = [
      `Architect needs you`,
      '',
      `Task: ${opts.workItemTitle}`,
      `Plan: ${opts.planSummary.slice(0, 200)}`,
      '',
      `Changes: ${opts.diffStats}`,
      '',
      `Approve: ${this.deps.callbackBaseUrl}/api/admin/architect/callbacks/${approveToken}`,
      `Reject: ${this.deps.callbackBaseUrl}/api/admin/architect/callbacks/${rejectToken}`,
      '',
      `Or review in dashboard: ${this.deps.dashboardBaseUrl}/architect/cycles/${opts.cycleId}`,
    ].join('\n')

    await this.deps.broadcast(message)
  }

  async notifyDigest(opts: {
    merged: number
    opened: number
    rejected: number
    errors: number
    openPrUrl?: string
  }): Promise<void> {
    const lines = [
      `Architect daily digest`,
      '',
      `Merged: ${opts.merged} PRs`,
      `Opened: ${opts.opened} PRs awaiting review`,
      `Rejected: ${opts.rejected} plans`,
      `Errors: ${opts.errors}`,
    ]
    if (opts.openPrUrl) lines.push('', `Review: ${opts.openPrUrl}`)
    await this.deps.broadcast(lines.join('\n'))
  }
}
```

// packages/core/src/api/routes/architect-callbacks.ts
import type { FastifyInstance } from 'fastify'
import type { CycleStore } from '../../architect/cycle-store.js'

export interface CallbackRouteDeps {
  cycleStore: CycleStore
  verifyToken: (token: string) => { cycleId: string; gate: string; decision: string; expiresAt: number } | null
}

export async function architectCallbackRoutes(app: FastifyInstance, deps: CallbackRouteDeps): Promise<void> {
  // POST /api/admin/architect/callbacks/:token — resolve an approval gate
  app.post<{ Params: { token: string } }>('/api/admin/architect/callbacks/:token', async (req, reply) => {
    const payload = deps.verifyToken(req.params.token)
    if (!payload) return reply.code(410).send({ error: 'expired_or_invalid', message: 'This link has expired or is invalid.' })

    deps.cycleStore.recordApproval(payload.cycleId, {
      gate: payload.gate,
      decision: payload.decision,
      decidedBy: 'callback',
    })

    return { ok: true, decision: payload.decision }
  })

  // Also support GET for simple link clicks
  app.get<{ Params: { token: string } }>('/api/admin/architect/callbacks/:token', async (req, reply) => {
    const payload = deps.verifyToken(req.params.token)
    if (!payload) return reply.code(410).send({ error: 'expired_or_invalid' })

    deps.cycleStore.recordApproval(payload.cycleId, {
      gate: payload.gate,
      decision: payload.decision,
      decidedBy: 'callback',
    })

    return reply.type('text/html').send(`<h1>Done</h1><p>Decision: ${payload.decision}</p><p><a href="/">Back to dashboard</a></p>`)
  })
}
```

// packages/core/src/api/routes/architect-events-sse.ts
import type { FastifyInstance } from 'fastify'
import type Database from 'better-sqlite3'

export interface ArchitectSSEDeps {
  db: Database.Database
}

export async function architectSSERoutes(app: FastifyInstance, deps: ArchitectSSEDeps): Promise<void> {
  app.get('/api/admin/architect/events', async (req, reply) => {
    reply.raw.setHeader('Content-Type', 'text/event-stream')
    reply.raw.setHeader('Cache-Control', 'no-cache')
    reply.raw.setHeader('Connection', 'keep-alive')
    reply.raw.setHeader('X-Accel-Buffering', 'no')
    reply.raw.flushHeaders()

    let lastId = 0
    const since = Number((req.query as any)?.since ?? 0)
    if (since > 0) lastId = since

    // Send initial batch
    const initial = deps.db.prepare(
      'SELECT * FROM architect_events WHERE created_at > ? ORDER BY created_at ASC LIMIT 50'
    ).all(lastId) as any[]
    for (const event of initial) {
      reply.raw.write(`data: ${JSON.stringify(event)}\n\n`)
      lastId = Math.max(lastId, event.created_at)
    }

    // Poll every 2 seconds for new events
    const interval = setInterval(() => {
      const newEvents = deps.db.prepare(
        'SELECT * FROM architect_events WHERE created_at > ? ORDER BY created_at ASC LIMIT 20'
      ).all(lastId) as any[]
      for (const event of newEvents) {
        reply.raw.write(`data: ${JSON.stringify(event)}\n\n`)
        lastId = Math.max(lastId, event.created_at)
      }
    }, 2000)

    req.raw.on('close', () => clearInterval(interval))
    await new Promise<void>(resolve => req.raw.on('close', resolve))
  })
}
```
